1. Apêndice: Matrizes HRBF

Foi visto na página principal deste relatório que o seguinte sistema deve ser resolvido para encontrar os coeficientes da função interpoladora: \begin{eqnarray} & \sum_{j=1}^{n}\begin{Bmatrix} \alpha_{j}\psi\left(\textbf{x}_{i} - \textbf{x}_{j} \right) - \left\langle \beta_{j},\nabla\psi \left(\textbf{x}_{i} - \textbf{x}_{j} \right) \right \rangle \end{Bmatrix} + \sum_{j=1}^{N}\lambda_{j}p_{j}\left(\textbf{x}_i\right) = 0, \quad i = 1, \ldots, n, \label{EqHRBF1}\\ & \sum_{j=1}^{n}\begin{Bmatrix} \alpha_{j}\nabla\psi\left(\textbf{x}_{i} -\textbf{x}_{j} \right) - \text{H}\psi\left(\textbf{x}_{i} - \textbf{x}_{j} \right)\beta_{j} \end{Bmatrix} + \sum_{j=1}^{N}\lambda_{j}\nabla p_{j}\left(\textbf{x}_i\right) =\textbf{n}_{i}, \quad i = 1, \ldots, n, \label{EqHRBF2} \\ & \sum_{j=1}^{n}\begin{Bmatrix} \alpha_{j}p_{k}\left(\textbf{x}_{j} \right) - \left\langle \beta_{j},\nabla p_{k} \left(\textbf{x}_{j} \right) \right \rangle \end{Bmatrix}=0, \quad k = 1,\ldots,N, \label{EqHRBF3} \end{eqnarray} em que $\text{H}$ é o operador Hessiano.

Vimos também que este sistema pode ser escrito em uma forma matricial por blocos como: \begin{equation} \overset{ \begin{array}{lrr} \hspace{6.0mm} {\color{red}n} & \hspace{4.5mm} \color{red}{3n} & \hspace{4.5mm}{\color{red}N} \end{array} } { \begin{array}{r} {\scriptsize \color{red}{n}} \\ {\scriptsize \color{red}{3n}} \\ {\scriptsize \color{red}{N}} \end{array} \begin{bmatrix} \textbf{A} & \textbf{-D} & \textbf{P} \\ \textbf{D}_2 & \textbf{-G} & \textbf{P}_{G} \\ \textbf{A}_{P} & \textbf{G}_{P} & 0 \end{bmatrix} } \begin{bmatrix} \alpha\\ \beta\\ \lambda \end{bmatrix} = \begin{bmatrix} \textbf{0}\\ \textbf{n}\\ \textbf{0} \end{bmatrix}, \label{Eq10} \end{equation} em que \begin{split} & \alpha = \left [ \alpha_{1} \quad \alpha_{2} \quad \alpha_{3} \dots \alpha_{n}\right ]^{T} \in \mathbb{R}^{n}, \\ & \beta = \left [ \beta_{1} \quad \beta_{2} \quad \beta_{3} \dots \beta_{n}\right ] = \left [ \beta_{1}^{x} \quad \beta_{1}^{y} \quad \beta_{1}^{z} \quad \beta_{2}^{x} \quad \beta_{2}^{y} \quad \beta_{2}^{z} \quad \dots \quad \beta_{n}^{x} \quad \beta_{n}^{y} \quad \beta_{n}^{z}\right ] \in \mathbb{R}^{3n}, \\ & \lambda = \left [ \lambda_{1} \quad \lambda_{2} \quad \lambda_{3} \quad \dots \quad \lambda_{n}\right ] \in \mathbb{R}^n, \end{split} e os valores em vermelho indicam as dimensões de cada bloco.

Abaixo são descritos com mais detalhes a forma de cada um dos blocos que compõem a matriz geral do sistema \eqref{Eq10}.

\psi\left(\textbf{x}_{i} - \textbf{x}_{j}\right)$. \begin{equation} \textbf{D}_1=\begin{bmatrix} \nabla\psi_{11} & \nabla\psi_{12} & \cdots & \nabla\psi_{1n} \\ \nabla\psi_{21} & \nabla\psi_{22} & \cdots & \nabla\psi_{2n} \\ \vdots & \vdots & \vdots & \ddots \\ \nabla\psi_{n1} & \nabla\psi_{n2} & \cdots & \nabla\psi_{nn} \end{bmatrix}_{n\times 3n}, \end{equation} em que $\nabla\psi_{ij} = \left[\frac{\partial \psi_{ij}}{\partial x} \ \frac{\partial \psi_{ij}}{\partial y} \ \frac{\partial \psi_{ij}}{\partial z}\right]$ é um vetor linha. \begin{equation} \textbf{P} = \begin{bmatrix} p_{1}(\textbf{x}_{1}) & p_{2}(\textbf{x}_{1}) & \cdots & p_{N}(\textbf{x}_{1}) \\ p_{1}(\textbf{x}_{2}) & p_{2}(\textbf{x}_{2}) & \cdots & p_{N}(\textbf{x}_{2}) \\ \vdots & \vdots & \vdots& \ddots \\ p_{1}(\textbf{x}_{n}) & p_{2}(\textbf{x}_{n}) & \cdots & p_{N}(\textbf{x}_{n}) \end{bmatrix}_{n\times N}, \end{equation} \begin{equation} \textbf{D}_2=\begin{bmatrix} (\nabla\psi_{11})^T & (\nabla\psi_{12})^T & \cdots & (\nabla\psi_{1n})^T \\ (\nabla\psi_{21})^T & (\nabla\psi_{22})^T & \cdots & (\nabla\psi_{2n})^T \\ \vdots & \vdots & \vdots & \ddots \\ (\nabla\psi_{n1})^T & (\nabla\psi_{n2})^T & \cdots & (\nabla\psi_{nn})^T \end{bmatrix}_{3n\times n}, \end{equation} em que $\nabla\psi_{ij} = \left[\frac{\partial \psi_{ij}}{\partial x} \ \frac{\partial \psi_{ij}}{\partial y} \ \frac{\partial \psi_{ij}}{\partial z}\right]$ é um vetor linha. \begin{equation} \textbf{G} = \begin{bmatrix} \textbf{H}\psi_{11} & \textbf{H}\psi_{12} & \cdots & \textbf{H}\psi_{1n} \\ \textbf{H}\psi_{21} & \textbf{H}\psi_{22} & \cdots & \textbf{H}\psi_{2n} \\ \vdots & \vdots & \vdots & \ddots \\ \textbf{H}\psi_{n1} & \textbf{H}\psi_{n2} & \cdots & \textbf{H}\psi_{nn} \end{bmatrix}_{3n\times 3n}, \end{equation} \begin{equation} \textbf{P}_{G}=\begin{bmatrix} (\nabla p_1(\textbf{x}_1))^T & (\nabla p_2(\textbf{x}_1))^T & \cdots & (\nabla p_N(\textbf{x}_1))^T \\ p_{N}(\textbf{x}_{n}) \end{bmatrix}_{N\times n}, \end{equation} \begin{equation} \textbf{G}_{P}=\begin{bmatrix} \nabla p_1(\textbf{x}_1) & \nabla p_1(\textbf{x}_2) & \cdots & \nabla p_1(\textbf{x}_n) \\ \nabla p_2(\textbf{x}_1) & \nabla p_2(\textbf{x}_2) & \cdots & \nabla p_2(\textbf{x}_n)) \\ \vdots & \vdots & \vdots & \ddots \\ \nabla p_N(\textbf{x}_1) & \nabla p_N(\textbf{x}_1) & \cdots &
